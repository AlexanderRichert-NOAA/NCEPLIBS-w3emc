61 SUBROUTINE w3ft11(VLN,GN,PLN,EPS,FL,WORK,TRIGS,RCOS)
62C
63 COMPLEX FL( 31 )
64 COMPLEX VLN( 32 , 31 )
65C
66 REAL COLRA
67 REAL EPS( 992 )
68 REAL GN(145,37)
69 REAL PLN( 32 , 31 )
70 REAL RCOS(37)
71 REAL TRIGS(216)
72 REAL WORK(144)
73C
74 SAVE
75C
76 DATA pi /3.14159265/
77C
78 drad = 2.5 * pi / 180.0
79C
80 DO 400 lat = 2,37
81 colra = (lat-1) * drad
82 CALL w3fa12(pln,colra, 30 ,eps)
83C
84 DO 100 l = 1, 31
85 fl(l) = (0.,0.)
86 100 CONTINUE
87C
88 DO 300 l = 1, 31
89C
90 DO 200 i = 1, 31 ,2
91 fl(l) = fl(l)+cmplx(pln(i,l) * real(vln(i,l)) ,
92 & pln(i,l) * aimag(vln(i,l)) )
93 fl(l) = fl(l)-cmplx(pln(i+1,l) * real(vln(i+1,l)),
94 & pln(i+1,l) * aimag(vln(i+1,l)))
95 200 CONTINUE
96C
97 fl(l) = cmplx(real(fl(l))*rcos(lat),aimag(fl(l))*rcos(lat))
98C
99 300 CONTINUE
100C
101 CALL w3ft12(fl,work,gn(1,lat ),trigs)
102C
103 400 CONTINUE
104C
105C*** POLE ROW = CLOSEST LATITUDE ROW
106C
107 DO 500 i = 1,145
108 gn(i,1) = gn(i,2)
109 500 CONTINUE
110 RETURN
111 END

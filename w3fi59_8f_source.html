3 C> SUBPROGRAM: W3FI59 FORM AND PACK POSITIVE, SCALED DIFFERENCES
4 C> PRGMMR: ALLARD, R. ORG: NMC41 DATE: 84-08-01
5 C>
6 C> ABSTRACT: CONVERTS AN ARRAY OF SINGLE PRECISION REAL NUMBERS INTO
7 C> AN ARRAY OF POSITIVE SCALED DIFFERENCES (NUMBER(S) - MINIMUM VALUE),
8 C> IN INTEGER FORMAT AND PACKS THE ARGUMENT-SPECIFIED NUMBER OF
9 C> SIGNIFICANT BITS FROM EACH DIFFERENCE.
10 C>
11 C> PROGRAM HISTORY LOG:
12 C> 84-08-01 ALLARD ORIGINAL AUTHOR
13 C> 90-05-17 R.E.JONES CONVERT TO CRAY CFT77 FORTRAN
14 C> 90-05-18 R.E.JONES CHANGE NAME PAKMAG TO W3LIB NAME W3FI59
15 C> 93-07-06 R.E.JONES ADD NINT TO DO LOOP 2000 SO NUMBERS ARE
16 C> ROUNDED TO NEAREST INTEGER, NOT TRUNCATED.
17 C> 94-01-05 IREDELL COMPUTATION OF ISCALE FIXED WITH RESPECT TO
18 C> THE 93-07-06 CHANGE.
19 C> 98-06-30 EBISUZAKI LINUX PORT
20 C>
21 C> USAGE: CALL W3FI59(FIELD,NPTS,NBITS,NWORK,NPFLD,ISCALE,LEN,RMIN)
22 C> INPUT ARGUMENT LIST:
23 C> FIELD - ARRAY OF FLOATING POINT DATA FOR PROCESSING (REAL)
24 C> NPTS - NUMBER OF DATA VALUES TO PROCESS IN FIELD (AND NWORK)
25 C> WHERE, NPTS > 0
26 C> NBITS - NUMBER OF SIGNIFICANT BITS OF PROCESSED DATA TO BE PACKED
27 C> WHERE, 0 < NBITS < 32+1
28 C>
29 C> OUTPUT ARGUMENT LIST:
30 C> NWORK - ARRAY FOR INTEGER CONVERSION (INTEGER)
31 C> IF PACKING PERFORMED (SEE NOTE BELOW), THE ARRAY WILL
32 C> CONTAIN THE PRE-PACKED, RIGHT ADJUSTED, SCALED, INTEGER
33 C> DIFFERENCES UPON RETURN TO THE USER.
34 C> (THE USER MAY EQUIVALENCE FIELD AND NWORK. SAME SIZE.)
35 C> NPFLD - ARRAY FOR PACKED DATA (character*1)
36 C> (DIMENSION MUST BE AT LEAST (NBITS * NPTS) / 64 + 1 )
37 C> ISCALE- POWER OF 2 FOR RESTORING DATA, SUCH THAT
38 C> DATUM = (DIFFERENCE * 2**ISCALE) + RMIN
39 C> LEN - NUMBER OF PACKED BYTES IN NPFLD (SET TO 0 IF NO PACKING)
40 C> WHERE, LEN = (NBITS * NPTS + 7) / 8 WITHOUT REMAINDER
41 C> RMIN - MINIMUM VALUE (REFERENCE VALUE SUBTRACTED FROM INPUT DATA)
42 C> THIS IS A CRAY FLOATING POINT NUMBER, IT WILL HAVE TO BE
43 C> CONVERTED TO AN IBM370 32 BIT FLOATING POINT NUMBER AT
44 C> SOME POINT IN YOUR PROGRAM IF YOU ARE PACKING GRIB DATA.
45 C>
46 C> REMARKS: LEN = 0 AND NO PACKING PERFORMED IF
47 C>
48 C> (1) RMAX = RMIN (A CONSTANT FIELD)
49 C> (2) NBITS VALUE OUT OF RANGE (SEE INPUT ARGUMENT)
50 C> (3) NPTS VALUE LESS THAN 1 (SEE INPUT ARGUMENT)
51 C>
52 C> ATTRIBUTES:
53 C> LANGUAGE: CRAY CFT77 FORTRAN
54 C> MACHINE: CRAY C916/256, Y-MP8/864, Y-MP EL92/256, J916/2048
55 C>
56  SUBROUTINE w3fi59(FIELD,NPTS,NBITS,NWORK,NPFLD,ISCALE,LEN,RMIN)
57 C NATURAL LOGARITHM OF 2 AND 0.5 PLUS NOMINAL SAFE EPSILON
58  parameter(alog2=0.69314718056,hpeps=0.500001)
59 C
60  REAL FIELD(*)
61 C
62  CHARACTER*1 NPFLD(*)
63  INTEGER NWORK(*)
64 C
65  DATA kzero / 0 /
66 C
67 C / / / / / /
68 C
69  len = 0
70  iscale = 0
71  IF (nbits.LE.0.OR.nbits.GT.32) GO TO 3000
72  IF (npts.LE.0) GO TO 3000
73 C
74 C FIND THE MAX-MIN VALUES IN FIELD.
75 C
76  rmax = field(1)
77  rmin = rmax
78  DO 1000 k = 2,npts
79  rmax = amax1(rmax,field(k))
80  rmin = amin1(rmin,field(k))
81  1000 CONTINUE
82 C
83 C IF A CONSTANT FIELD, RETURN WITH NO PACKING PERFORMED AND 'LEN' = 0.
84 C
85  IF (rmax.EQ.rmin) GO TO 3000
86 C
87 C DETERMINE LARGEST DIFFERENCE IN FIELD (BIGDIF).
88 C
89  bigdif = rmax - rmin
90 C
91 C ISCALE IS THE POWER OF 2 REQUIRED TO RESTORE THE PACKED DATA.
92 C ISCALE IS COMPUTED AS THE LEAST INTEGER SUCH THAT
93 C BIGDIF*2**(-ISCALE) < 2**NBITS-0.5
94 C IN ORDER TO ENSURE THAT THE PACKED INTEGERS (COMPUTED IN LOOP 2000
95 C WITH THE NEAREST INTEGER FUNCTION) STAY LESS THAN 2**NBITS.
96 C
97  iscale=nint(alog(bigdif/(2.**nbits-0.5))/alog2+hpeps)
98 C
99 C FORM DIFFERENCES, RESCALE, AND CONVERT TO INTEGER FORMAT.
100 C
101  twon = 2.0 ** (-iscale)
102  DO 2000 k = 1,npts
103  nwork(k) = nint( (field(k) - rmin) * twon )
104  2000 CONTINUE
105 C
106 C PACK THE MAGNITUDES (RIGHTMOST NBITS OF EACH WORD).
107 C
108  koff = 0
109  iskip = 0
110 C
111 C USE NCAR ARRAY BIT PACKER SBYTES (GBYTES PACKAGE)
112 C
113  CALL sbytesc(npfld,nwork,koff,nbits,iskip,npts)
114 C
115 C ADD 7 ZERO-BITS AT END OF PACKED DATA TO INSURE BYTE BOUNDARY.
116 C USE NCAR WORD BIT PACKER SBYTE
117 C
118  noff = nbits * npts
119  CALL sbytec(npfld,kzero,noff,7)
120 C
121 C DETERMINE BYTE LENGTH (LEN) OF PACKED FIELD (NPFLD).
122 C
123  len = (noff + 7) / 8
124 C
125  3000 CONTINUE
126  RETURN
127 C
128  END

3C> SUBPROGRAM: W3FT07 TRANSFORM GRIDPOINT FLD BY INTERPOLATION
4C> PRGMMR: LIN ORG: NMC412 DATE:93-03-24
5C>
6C> ABSTRACT: TRANSFORMS DATA CONTAINED IN A GIVEN GRID ARRAY
7C> BY TRANSLATION, ROTATION ABOUT A COMMON POINT AND DILATATION
8C> IN ORDER TO CREATE A NEW GRID ARRAY ACCORDING TO SPECS.
9C>
10C> PROGRAM HISTORY LOG:
11C> 74-09-01 ORIGINAL AUTHOR(S): J. MCDONELL, J.HOWCROFT
12C> 84-06-27 R.E.JONES CHANGE TO IBM VS FORTRAN
13C> 89-01-24 R.E.JONES CHANGE TO MICROSOFT FORTRAN 4.10
14C> 89-03-31 R.E.JONES CHANGE TO VAX-11 FORTRAN
15C> 93-03-16 D. SHIMOMURA -- RENAMED FROM W3FT00() TO W3FT07()
16C> IN ORDER TO MAKE MINOR MODS WHILE DOING F77.
17C> CHANGES TO CALL SEQUENCE; CHANGES TO VRBL NAMES;
18C> ADDED COMMENTS.
19C>
20C> ... 1 2 3 4 5 6 7 8
21C> USAGE: CALL W3FT07(FLDA,IA,JA,AIPOLE,AJPOLE,BIPOLE,BJPOLE,DSCALE,
22C> ANGLE,LINEAR,LDEFQQ,DEFALT,FLDB,IB,JB)
23C> 9 10 11 12 13 14 15
24C> INPUT ARGUMENT LIST:
25C> FLDA(IA,JA) - REAL*4 ORIGINAL SOURCE GRID-POINT DATA FIELD
26C> AIPOLE,AJPOLE - REAL*4 COMMON POINT I- AND J-COORDINATES OF THE
27C> ORIGINAL FIELD, ASSUMING A RIGHT-HAND CARTESIAN
28C> COORDINATE SYSTEM. THE POINT NEED NOT BE INSIDE
29C> THE BOUNDS OF EITHER GRID
30C> AND CAN HAVE FRACTIONAL VALUES.
31C> COMMON POINT ABOUT WHICH TO ROTATE THE GRIDPOINTS
32C> BIPOLE,BJPOLE - REAL*4 COMMON POINT I- AND J-COORDINATES FOR
33C> TRANSFORMED DESTINATION GRID
34C> DSCALE - REAL*4 SCALE-CHANGE (DILATION) EXPRESSED AS
35C> A RATIO OF THE TRANSFORMED FIELD TO THE ORIGINAL
36C> FIELD
37C> DSCALE = GRDLENKM(DESTINATION) / GRDLENKM(SOURCE)
38C>
39C> ANGLE - REAL*4 DEGREE MEASURE OF THE ANGLE REQUIRED TO
40C> ROTATE THE J-ROW OF THE ORIGINAL GRID INTO
41C> COINCIDENCE WITH THE NEW GRID. (+ COUNTER-
42C> CLOCKWISE, - CLOCKWISE)
43C> ANGLE = VERTLONW(SOURCE) - VERTLONW(DESTINATION)
44C>
45C> LINEAR - LOGICAL*4 INTERPOLATION-METHOD SELECTION SWITCH:
46C> .TRUE. BI-LINEAR INTERPOLATION
47C> .FALSE. BI-QUADRATIC INTERPOLATION
48C>
49C> LDEFQQ - LOGICAL*4 DEFAULT-VALUE SWITCH:
50C> IF .TRUE. THEN
51C> USE DEFAULT-VALUE FOR DESTINATION POINT
52C> OUT-OF-BOUNDS OF GIVEN GRID;
53C> ELSE
54C> EXTRAPOLATE COARSELY FROM NEARBY BNDRY POINT
55C>
56C> DEFALT - REAL*4 THE DEFAULT-VALUE TO USE IF LDEFQQ = .TRUE.
57C>
58C> OUTPUT ARGUMENT LIST:
59C> FLDB(IB,JB) - REAL*4 RESULTING TRANSFORMED DESTINATION FIELD
60C>
61C>
62C> REMARKS: LIST CAVEATS, OTHER HELPFUL HINTS OR INFORMATION
63C> IN GENERAL 'FLDA' AND 'FLDB' CANNOT BE EQUIVALENCED
64C> ALTHOUGH THERE ARE SITUATIONS IN WHICH IT WOULD BE SAFE TO DO
65C> SO. CARE SHOULD BE TAKEN THAT ALL OF THE NEW GRID POINTS LIE
66C> WITHIN THE ORIGINAL GRID, NO ERROR CHECKS ARE MADE.
67C>
68C>

70C> LANGUAGE: CRAY CFT77 FORTRAN 77

73 SUBROUTINE w3ft07(FLDA,IA,JA,AIPOLE,AJPOLE,BIPOLE,BJPOLE,
74 A DSCALE,ANGLE,LINEAR,LDEFQQ,DEFALT,FLDB,IB,JB)
75C
76 REAL FLDA(IA,JA)
77 REAL AIPOLE,AJPOLE
78 REAL BIPOLE,BJPOLE
79 REAL DSCALE
80 REAL ANGLE
81 REAL DEFALT
82 REAL FLDB(IB,JB)
83 REAL ERAS(4)
84 REAL TINY
85C
86 LOGICAL LINEAR
87 LOGICAL LDEFQQ
88C
89 SAVE
90C
91 DATA tiny / 0.001 /
92C
93C ... WHERE TINY IS IN UNITS OF 1.0 = 1 GRID INTERVAL
94C
95C . . . . . S T A R T . . . . . . . . . . . . . . . . . . .
96C
97 theta = angle * (3.14159/180.)
98 sint = sin(theta)
99 cost = cos(theta)
100C
101C ... WE WILL SCAN ALONG THE J-ROW OF THE DESTINATION GRID ...
102 DO 288 jn = 1,jb
103 brelj = float(jn) - bjpole
104C
105 DO 277 in = 1,ib
106 breli = float(in) - bipole
107 sti = aipole + dscale*(breli*cost - brelj*sint)
108 stj = ajpole + dscale*(breli*sint + brelj*cost)
109 im = sti
110 jm = stj
111C
112C ... THE PT(STI,STJ) IS THE LOCATION OF THE FLDB(IN,JN)
113C ... IN FLDA,S COORDINATE SYSTEM
114C ... IS THIS POINT LOCATED OUTSIDE FLDA?
115C ... ON THE BOUNDARY LINE OF FLDA?
116C ... ON THE FIRST INTERIOR GRIDPOINT OF FLDA?
117C ... GOOD INSIDER, AT LEAST 2 INTERIOR GRIDS INSIDE?
118 ioff = 0
119 joff = 0
120 kquad = 0
121C
122 IF (im .LT. 1) THEN
123C ... LOCATED OUTSIDE OF FLDA, OFF LEFT SIDE ...
124 ii = 1
125 ioff = 1
126 ELSE IF (im .EQ. 1) THEN
127C ... LOCATED ON BOUNDARY OF FLDA, ON LEFT EDGE ...
128 kquad = 5
129 ELSE
130C ...( IM .GT. 1) ... LOCATED TO RIGHT OF LEFT-EDGE ...
131 IF ((ia-im) .LT. 1) THEN
132C ... LOCATED OUTSIDE OF OR EXACTLY ON RIGHT EDGE OF FLDA ..
133 ii = ia
134 ioff = 1
135 ELSE IF ((ia-im) .EQ. 1) THEN
136C ... LOCATED ON FIRST INTERIOR PT WITHIN RIGHT EDGE OF FLDA
137 kquad = 5
138 ELSE
139C ... (IA-IM) IS .GT. 1) ...GOOD INTERIOR, AT LEAST 2 INSIDE
140 ENDIF
141 ENDIF
142C
143C . . . . . . . . . . . . . . .
144C
145 IF (jm .LT. 1) THEN
146C ... LOCATED OUTSIDE OF FLDA, OFF BOTTOM ...
147 jj = 1
148 joff = 1
149 ELSE IF (jm .EQ. 1) THEN
150C ... LOCATED ON BOUNDARY OF FLDA, ON BOTTOM EDGE ...
151 kquad = 5
152 ELSE
153C ...( JM .GT. 1) ... LOCATED ABOVE BOTTOM EDGE ...
154 IF ((ja-jm) .LT. 1) THEN
155C ... LOCATED OUTSIDE OF OR EXACTLY ON TOP EDGE OF FLDA ..
156 jj = ja
157 joff = 1
158 ELSE IF ((ja-jm) .EQ. 1) THEN
159C ... LOCATED ON FIRST INTERIOR PT WITHIN TOP EDGE OF FLDA
160 kquad = 5
161 ELSE
162C ... ((JA-JM) .GT. 1) ...GOOD INTERIOR, AT LEAST 2 INSIDE
163 ENDIF
164 ENDIF
165C
166 IF ((ioff + joff) .EQ. 0) THEN
167 GO TO 244
168 ELSE IF ((ioff + joff) .EQ. 2) THEN
169 GO TO 233
170 ENDIF
171C
172 IF (ioff .EQ. 1) THEN
173 jj = stj
174 ENDIF
175 IF (joff .EQ. 1) THEN
176 ii = sti
177 ENDIF
178 233 CONTINUE
179 IF (ldefqq) THEN
180 fldb(in,jn) = defalt
181 ELSE
182 fldb(in,jn) = flda(ii,jj)
183 ENDIF
184 GO TO 277
185C
186C . . . . . . . . . . . . .
187C
188 244 CONTINUE
189 i = sti
190 j = stj
191 xdeli = sti - float(i)
192 xdelj = stj - float(j)
193C
194 IF ((abs(xdeli) .LT. tiny) .AND. (abs(xdelj) .LT. tiny)) THEN
195C ... THIS POINT IS RIGHT AT A GRIDPOINT. NO INTERP NECESSARY
196 fldb(in,jn) = flda(i,j)
197 GO TO 277
198 ENDIF
199C
200 IF ((kquad .EQ. 5) .OR. (linear)) THEN
201C ... PERFORM BI-LINEAR INTERP ...
202 eras(1) = flda(i,j)
203 eras(4) = flda(i,j+1)
204 eras(2) = eras(1) + xdeli*(flda(i+1,j) - eras(1))
205 eras(3) = eras(4) + xdeli*(flda(i+1,j+1) - eras(4))
206 di = eras(2) + xdelj*(eras(3) - eras(2))
207 GO TO 266
208C
209 ELSE
210C ... PERFORM BI-QUADRATIC INTERP ...
211 xi2tm = xdeli * (xdeli-1.) * 0.25
212 xj2tm = xdelj * (xdelj-1.) * 0.25
213 j1 = j - 1
214 DO 255 k=1,4
215 eras(k)=(flda(i+1,j1)-flda(i,j1))*xdeli+flda(i,j1)+
216 a (flda(i-1,j1)-flda(i,j1)-flda(i+1,j1)+flda(i+2,j1))*xi2tm
217 j1 = j1 + 1
218 255 CONTINUE
219C
220 di = eras(2) + xdelj*(eras(3)-eras(2)) +
221 a xj2tm*(eras(4)-eras(3)-eras(2)+eras(1))
222 GO TO 266
223 ENDIF
224C
225 266 CONTINUE
226 fldb(in,jn) = di
227 277 CONTINUE
228 288 CONTINUE
229C
230 RETURN
231 END
